## Usage

```jsx
const items: TabsProps["items"] = [
    {
      key: "1",
      label: "Overview",
      children: (
        <>
          <span>Content of Overview</span>
        </>
      ),
    },
    {
      key: "2",
      label: "Riwayat Transaksi",
      children: (
        <>
          <span>Content of Riwayat Transaksi</span>
        </>
      ),
    },
]

<Tabs defaultActiveKey="1" items={items} onChange={onChange} />
```
