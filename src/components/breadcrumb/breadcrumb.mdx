## Usage

```jsx
const items: BreadcrumbProps["items"] = [
    {
      title: <a>Transaksi</a>,
    },
    {
      title: <a>Daftar Transaksi</a>,
    },
    {
      title: "TRF-081-2230KlO91",
    },
  ]

<Breadcrumb items={items} />
```
