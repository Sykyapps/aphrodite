## Usage

```jsx
<Form onFinish={onFinish} onChange={onChange}>
  <FormItem
    name="email"
    label="Email"
    rules={[
      { required: true, message: "Email wajib diisi." },
      { type: "email", message: "Format email tidak sesuai." },
    ]}
  >
    <Input placeholder="Masukkan email" />
  </FormItem>
  <FormItem
    name="password"
    label="Password"
    rules={[{ required: true, message: "Password wajib diisi." }]}
  >
    <InputPassword placeholder="Masukkan password" />
  </FormItem>
  <FormItem name="confirmPassword" label="Konfirmasi Password">
    <InputPassword placeholder="Masukkan konfirmasi password" />
  </FormItem>
  <Button htmlType="submit">Submit</Button>
</Form>
```
