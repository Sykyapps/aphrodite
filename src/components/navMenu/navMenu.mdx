## Usage

```jsx
const items: NavMenuProps["items"] = [
  {
    label: "Home",
    key: "home",
  },
  {
    label: "Settings",
    key: "settings",
    children: [
      {
        label: "Account Settings",
        key: "accountSettings",
      },
    ],
  },
  {
    label: "About",
    key: "about",
    disabled: true,
  },


<NavMenu items={items} defaultSelectedKeys={["home"]} selectedKeys={["home"]} />
```
