## Usage

```jsx
const { notification } = ContextProvider.useApp()

const handleOpenNotification = () => {
  showNotification({
    notification,
    type: "success",
    message: "Message",
  })
}

return <Button onClick={handleOpenNotification}>Show Notification</Button>
```
